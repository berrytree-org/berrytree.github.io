{{/* breadcrumb shortcode Renders a clear, accessible breadcrumb trail for wayfinding. Designed for elderly visitors: large text, obvious path. Parameters: trail (string) — pipe-separated breadcrumb items Format: "Label|URL|Label|URL|Current Label" Last item has no URL (current page) Usage: {{< breadcrumb trail="Home|/|Ancestors|/ancestors/|David J. Berry Sr." />}} */}}
    {{- $items := split (.Get "trail") "|" }} {{- $len := len $items }} {{- range $i, $item := $items }} {{- if eq (mod $i 2) 0 }} {{- if lt (add $i 1) $len }} {{- $url := index $items (add $i 1) }}
  1. {{ $item }}
    2. {{- else }}
  2. {{ $item }}
    3. {{- end }} {{- end }} {{- end }}
